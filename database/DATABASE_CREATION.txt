CREATE TABLE WORD (ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, WORD VARCHAR(48) NOT NULL);

CREATE TABLE LANGUAGE (ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, LANGUAGE VARCHAR(48) NOT NULL);

CREATE TABLE QUALITY (ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, WORD_ID INT NOT NULL, QUALITY INT NOT NULL CHECK (QUALITY=-1 OR QUALITY=0 OR QUALITY=1), FOREIGN KEY (WORD_ID) REFERENCES WORD(ID));

CREATE TABLE DIFFICULTY (ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, WORD_ID INT NOT NULL, DIFFICULTY INT NOT NULL CHECK (DIFFICULTY=1 OR DIFFICULTY=2 OR DIFFICULTY=3), FOREIGN KEY (WORD_ID) REFERENCES WORD(ID));

CREATE TABLE TRANSLATION (ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, TRANSLATION VARCHAR(48) NOT NULL, WORD_ID INT NOT NULL, LANGUAGE_ID INT NOT NULL, FOREIGN KEY (WORD_ID) REFERENCES WORD(ID), FOREIGN KEY (LANGUAGE_ID) REFERENCES LANGUAGE(ID));

CREATE TABLE RATE (ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, TRANSLATION_ID INT NOT NULL, RATE INT NOT NULL CHECK (RATE=1 OR RATE=2 OR RATE=3 OR RATE=4 OR RATE=5), FOREIGN KEY (TRANSLATION_ID) REFERENCES TRANSLATION(ID));
